Мультизагрузочная флешка (установка Grub2 из Ubuntu на USB)


1) Отформатировать флешку. Создать два раздела (FAT32 и NTFS)


2) Посмотреть список подключенных флешек


sudo fdisk -l


Возможно флешка будет отображаться как sdb

sdb - устройство

sdb1 - раздел на флешке (FAT32)

sdb2 - раздел на флешке (NTFS)

3) Создать папку для монтирования (например /mnt/USB)


sudo mkdir /mnt/USB


4) Смонтировать флешку в папку (автоматическое монтирование в media возможно не подходит, нужно проверить)


mount /dev/sdb1 /mnt/USB


В любом случае монтируем первый раздел флешки (желательно, чтобы он был в файловой системе FAT32)


5) grub-install --removable --boot-directory=/mnt/USB/boot --efi-directory=/mnt/USB/ /dev/sdb


Примечание: должно быть сообщение в терминале, что ошибок не обнаружено.


6) Отмонтировать флешку


umount /mnt/USB


**********************************************************************************

Этап создания файла для меню и копирования необходимых файлов (образов и прочего) на флешку


1) Узнать UUID всех разделов

ls -l /dev/disk/by-uuid


Уникальные метки разделов (15.01.2020):


FA232 E934-6B63

NTFS 15C6F8842A182E6C


2) Создать (или скопировать готовый) файл grub.cfg по пути /boot/grub/


**********************************************************************************

Дополнительные команды для терминала


grub-install --removable --target=x86_64-efi --boot-directory=/mnt/USB/boot --efi-directory=/mnt/USB/ /dev/sdb



grub-install --removable --boot-directory=/mnt/USB/boot --efi-directory=/mnt/USB/ /dev/sdb



grub-install --removable --debug --boot-directory=/mnt/USB/boot --target=x86_64-efi --efi-directory=/mnt/USB/ /dev/sdb


Команда для созадния карты устройств

sudo grub-mkdevicemap

# Поиск среди всех разделов компьютера нужного раздела (мультизагрузочной флешки)

# Меню для мультизагрузочной флешки


########################################

# Поиск среди всех разделов компьютера нужного раздела (мультизагрузочной флешки)

# по уникальному идентификатору UUID и фиксация данного раздела, как корневого


search -u E934-6B63 -s



# таймаут меню


set timeout=30


# выбранный пункт по умолчанию


set default=0


# разрешение


set gfxmode=1024x768,800x600


# загружаем шрифт и включаем графическое меню


loadfont /boot/polices/unicode.pf2

insmod gfxterm

insmod vbe

if terminal_output gfxterm ; then true ; else terminal gfxterm; fi

# настройка цветов


set color_normal=white/black

set menu_color_normal=white/black

set menu_color_highlight=light-blue/black


# Подключение библиотеки для работы с изображениями в формате png


insmod png


# Установка фонового изображения


background_image="/Background_image/image_1.png"


# Пункты загрузочного меню


menuentry "Reboot" {

insmod reboot

reboot

}


menuentry "ShutDown PC"{

halt

}


menuentry "Kubuntu (64 bit)" {

search -u E934-6B63 -s

set isofile="/Systems/Linux/kubuntu-18.04.3-desktop-amd64.iso"

loopback loop $isofile

linux (loop)/casper/vmlinuz boot=casper iso-scan/filename=${isofile}

initrd (loop)/casper/initrd

boot

}
