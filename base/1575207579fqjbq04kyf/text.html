def face(column, row):

""" Create a single face """

return (column * rows + row,

column * rows + 1 + row,

(column + 1) * rows + 1 + row,

(column + 1) * rows + row)

def vert(column, row, size):

""" Create a single vert """

return (column * size, row * size, 0)



import bpy

# Settings

name = 'Grid'

rows = 5

columns = 10


size = 0.3

faces = []

verts = [vert(x, y, size) for x in range(columns) for y in range(rows)]

faces = [face(x, y) for x in range(columns - 1) for y in range(rows - 1)]

# Create Mesh Datablock

mesh = bpy.data.meshes.new(name)

mesh.from_pydata(verts, [], faces)


# Create Object and link to scene

obj = bpy.data.objects.new(name, mesh)

#bpy.context.scene.objects.link(obj)

scene = bpy.context.scene

scene.collection.objects.link(obj)

#bpy.context.active_object


# Select the object


bpy.context.view_layer.objects.active = obj

#выбор объекта

bpy.context.scene.objects[0].select_set(True)
